224template<class T>
225int AVL<T>::insertaDato(NodoAVL<T>* &nodo, T &dato) {
226 NodoAVL<T> *p = nodo;
227 int deltaH = 0; //Variación de la altura del subárbol
228 if (!p) { //En la recursión se llega a un nodo hoja
229 p = new NodoAVL<T>(dato);
230 tam++;
231 nodo = p;
232 deltaH = 1;
233 } else if (dato > p->dato) {
234 if (insertaDato(p->der, dato)) {
235 p->bal--;
236 if (p->bal == -1)
237 deltaH = 1;
238 else if (p->bal == -2) {
239 if (p->der->bal == 1)
240 rotDecha(p->der); //Rotación doble caso 3
241 rotIzqda(nodo); //Rotación simple caso 4
242 }
243 }
244 } else if (dato < p->dato) {
245 if (insertaDato(p->izq, dato)) {
246 p->bal++;
247 if (p->bal == 1)
248 deltaH = 1;
249 else if (p->bal == 2) {
250 if (p->izq->bal == -1)
251 rotIzqda(p->izq); //Rotación doble caso 2
252 rotDecha(nodo); //Rotación simple caso 1
253 }
254 }
255 }
256 return deltaH;
257}
