8unsigned long THashImagen::hash(unsigned long &clave, unsigned long &intento) {
9 excpTablaCreada();
10 if(capacidadTabla == 0)
11 throw domain_error("[THashImagen::hash] No se puede calcular el hash con una tabla de capacidad 0.");
12
13 //La mayoría de datos son clave%2==0 (6032 elementos)
14 //La mayoría están entre 10^8 y 10^9. La minoría entre 10^5 y 10^6
15 unsigned long aux=1,aux2=1;
16
17 if(clave%2==0){ //Más que dispersar
18 for (int i = 0; i < 9; ++i)
19 aux = aux << 1; //Desplazo a la izqda el bit
20 aux = (aux - clave) % 100000000;
21 aux2 += clave % 1000000;
22 aux2 *= aux2;
23 }else{ //Menos que dispersar
24 unsigned long bucle = clave;
25 while (bucle){
26 aux2 += bucle%1000;
27 aux *= aux2 - aux;
28 bucle /= 1000;
29 }
30 }
31
32 while(!esPrimo(aux) && clave%aux == 0)
33 ++aux;
34 while(!esPrimo(aux2) && clave%aux2 == 0)
35 ++aux2;
36
37 unsigned long h1 = clave % aux, h2 = clave % aux2;
38 return (h1 + h2*intento) % capacidadTabla;
39}
