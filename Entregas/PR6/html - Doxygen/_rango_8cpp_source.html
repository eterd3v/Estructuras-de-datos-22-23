59 if(min.getLongitud() <= coords.getLongitud() && min.getLatitud() <= coords.getLatitud()) //Mayor o igual que min

80Rango Rango::interseccion(const Rango &rango) const{ //Un rango puede tener 0, 1, 2 o 4 coordenadas de otro rango
81 if(contiene(rango))
82 return rango;
83
84 UTM rMinAlt(rango.getMinX(),rango.getMaxY()); //vertice izquierdo superior
85 UTM rMaxAlt(rango.getMaxX(),rango.getMinY());
86
87 bool izqabj = this->contiene(rango.getMin()); //vertice de la izquierda inferior
88 bool derarr = this->contiene(rango.getMax()); //vertice de la derecha superior
89 bool izqarr = this->contiene(rMinAlt); //vertice de la izquierda superior
90 bool derabj = this->contiene(rMaxAlt); //vertice de la derecha inferior
91
92 //Mi rango incluye dos vertices del otro rango (es decir, una arista ya sea la izquierda, derecha, arriba o abajo)
93 if (izqabj & izqarr || derarr & derabj || derarr & izqarr || izqabj & derabj ){
94 Rango aux(rango);
95
96 if(izqabj & izqarr) //arista de la izquierda del rango => hay que desplazar la arista derecha
97 aux.setMin(UTM(this->getMin().getLongitud(),rango.getMin().getLatitud()));
98 if(derarr & derabj) //arista de la derecha del rango => hay que desplazar la arista izquierda
99 aux.setMax(UTM(this->getMax().getLongitud(),rango.getMax().getLatitud()));
100 if(derarr & izqarr) //arista de arriba del rango => hay que subir la arista inferior
101 aux.setMin(UTM(rango.getMin().getLongitud(),this->getMin().getLatitud()));
102 if(izqabj & derabj) //arista de abajo del rango => hay que bajar la arista superior
103 aux.setMax(UTM(rango.getMax().getLongitud(),this->getMax().getLatitud()));
104
105 return aux;
106 }
107
108 //Mi rango al menos contiene 1 UTM
109 if(izqabj || derarr || izqarr || derabj){
110 Rango aux(*this);
111 if(izqabj) //Contiene el vertice izquierdo inferior del rango.
112 aux.setMin(rango.getMin());
113 else if(derarr) //Contiene el vertice derecho superior del rango.
114 aux.setMax(rango.getMax());
115 else if(izqarr){ //Contiene el vertice izquierdo superior del rango.
116 aux.setMax(UTM(rMaxAlt.getLongitud(),max.getLatitud()));
117 aux.setMin(UTM(min.getLongitud(),rMaxAlt.getLatitud()));
118 }else{ //if(derabj) //Contiene el vertice derecho inferior del rango.
119 aux.setMin(UTM(rMaxAlt.getLongitud(),min.getLatitud()));
120 aux.setMax(UTM(max.getLongitud(),rMaxAlt.getLatitud()));
121 }
