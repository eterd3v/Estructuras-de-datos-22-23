113 void pintarPixel(Rango &limites, UTM &coords, RGBColor &color, bool grande = false){
114 if(limites.contiene(coords)){
115 double pixelPorGradoX = (double) (this->filas - 1) / (double) limites.distLati(); //La latitud en img.h aquí se entiene como la longitud
116 double pixelPorGradoY = (double) (this->columnas - 1) / (double) limites.distLong();
117
118 double vlon = (double) coords.getLongitud();
119 double vlat = (double) coords.getLatitud();
120
121 double minlat = limites.getMinY();
122 double minlon = limites.getMinX();
123
124 int posX = (vlat - minlat) * pixelPorGradoX;
125 int posY = this->columnas - 1 - ((vlon - minlon) * pixelPorGradoY);
126
127 if(grande)

161 void pintarElipse(Rango &limites, Rango &recuadro, RGBColor &color){
162 float semiejeA = recuadro.distLong() / 2; //Semieje mayor, a, de la elipse
163 float semiejeB = recuadro.distLati() / 2; //Semieje menor, b, de la elipse
164
165 float alfa = (float) M_PI / 2; //90 grados es el máximo que hace falta calcular. El resto es simétrico
166 float radian = 1 / (float) M_PI; //1 radián
167 float beta = radian / 20; //El aumento del ángulo, beta, determina la precisión. Cuanto menor, mejor
168
169 for (float i = 0.0; i <= alfa; i+=beta) { //Bucle para iterar los grados
170
171 float puntoX = semiejeA * cos(i); //Saco la coordenada X de la elipse
172 float puntoY = semiejeB * sin(i); //Saco la coordenada Y de la elipse
173
174 UTM utm1(recuadro.getCentro());
175 UTM utm2(recuadro.getCentro());
176 UTM utm3(recuadro.getCentro());
177 UTM utm4(recuadro.getCentro());
178
179 utm1.asignar(utm1.getLongitud() + puntoX, utm1.getLatitud() + puntoY); //Cuadrante 1
180 utm2.asignar(utm2.getLongitud() + puntoX, utm2.getLatitud() - puntoY); //Cuadrante 3
181 utm3.asignar(utm3.getLongitud() - puntoX, utm3.getLatitud() + puntoY); //Cuadrante 2
182 utm4.asignar(utm4.getLongitud() - puntoX, utm4.getLatitud() - puntoY); //Cuadrante 4
183
184 pintarPixel(limites,utm1,color); //Dibujamos los casos simétricos
185 pintarPixel(limites,utm2,color);
186 pintarPixel(limites,utm3,color);
187 pintarPixel(limites,utm4,color);
188 }
189
190 }
