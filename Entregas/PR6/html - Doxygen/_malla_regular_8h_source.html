134template<class T>
135Casilla<T>* MallaRegular<T>::obtenerCasilla(const float &x,const float &y){
136 int i = (x - rangoMalla.getMinX()) / tamCasilla.distLong();
137 int j = (y - rangoMalla.getMinY()) / tamCasilla.distLati();
138 if(rangoMalla.contiene(UTM(x,y)))
139 return &matriz[j][i]; //Filas es la j, Columnas es la i
140}

145template<class T>
146MallaRegular<T>::MallaRegular(const Rango &rangoMalla, const unsigned int &nDiv) : rangoMalla(rangoMalla), tamCasilla(), nCasillas(nDiv * nDiv){
147 float tamCasX = rangoMalla.distLong() / nDiv;
148 float tamCasY = rangoMalla.distLati() / nDiv;
149 tamCasilla.setMin(UTM(rangoMalla.getMinX() ,rangoMalla.getMaxY() - tamCasY)); //Coloco la casilla modelo en el mismo sitio en el que se podría recorrer la matriz
150 tamCasilla.setMax(UTM(rangoMalla.getMinX() + tamCasX,rangoMalla.getMaxY() ));
151 matriz.insert(matriz.begin(), nDiv, vector<Casilla<T> >(nDiv));
152};

162template<class T>
163vector<T> MallaRegular<T>::buscarRango(const Rango &r){
164 vector<T> aux;
165 try{ //Hay un método (intersección) puede provocar una excepción
166 Rango g = rangoMalla.interseccion(r); //Tomo la intersección del rango con el rango de la malla
167 for(int i = 0; i < matriz.size(); ++i)
168 for(int j = 0; j < matriz.size(); ++j)
169 for(auto it = matriz[i][j].puntos.begin(); it != matriz[i][j].puntos.end() ; ++it)
170 if(g.contiene((*it)->getPos())) //Si el dato está dentro de la intersección se añade
171 aux.push_back(*it);
172 }catch(exception &err){
173 cout << err.what() << endl;
174 }
175 return aux;
176}

180 unsigned int max = 0;

182 for(int j = 0; j < matriz.size(); ++j)
183 if(matriz[i][j].puntos.size() > max) //Guardo el máximo, que es el tamaño del contenedor

188template<class T>
189float MallaRegular<T>::promedioElementosPorCelda(){
190 unsigned int suma = 0;
191 for(int i = 0; i < matriz.size(); ++i)
192 for(int j = 0; j < matriz.size(); ++j)
193 suma += matriz[i][j].puntos.size(); //Sumo el tamaño que tiene cada lista
194 return (float) suma / (float) sqrt(nCasillas);
195}
