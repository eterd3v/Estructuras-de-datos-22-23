212template<class T>
213void VDinamico<T>::insertar(const T& dato, unsigned int pos){
214 if(tamLogico < pos && pos < UINT_MAX)
215 throw length_error("ERROR: No se puede insertar en esa posición ya que no está en el vector o sus límites.");
216
217 if(tamLogico >= tamFisico) //El tamaño se queda corto y hay que asignar más espacio para el vector
218 alojarDoble();
219
220 switch (pos) {
221 case UINT_MAX: tamLogico:
222 vectorDin[tamLogico]=dato;
223 break;
224 case 0:
225 throw invalid_argument("ERROR: La posición 0 no es válida para insertar un dato. Sólo entre 1 <= pos <= TAM");
226 break;
227 default:
228 T datoAux1, datoAux2 = dato;
229 for (unsigned int i = pos; i <= tamLogico; ++i) { //Se ejecuta hasta el tamLógico para copiar todos los auxiliares
230 datoAux1=vectorDin[i]; //Lo que haya en la posición, lo meto en un dato aux 1
231 vectorDin[i]=datoAux2; //Ahora se sobreescribe el dato con lo que hay en el dato aux 2
232 datoAux2 = datoAux1; //Como ya está en el vector, el siguiente dato que se va a almacenar será el dato aux 1 para desplazarse a la derecha
233 }
234 break;
235 }
236 tamLogico++;
237}

239template<class T>
240T VDinamico<T>::borrar(unsigned int pos){
241 if(tamLogico < pos && pos < UINT_MAX)
242 throw invalid_argument("ERROR: No se puede borrar en esa posición ya que no está en el vector o sus límites.");
243 if(tamLogico == 0)
244 throw invalid_argument("ERROR: No hay más elementos en el vector. No se puede borrar más.");
245 if(pos == 0)
246 throw invalid_argument("ERROR: La posición 0 no es válida para borrar un dato. Sólo entre 1 <= pos <= TAM");
247
248 if(tamLogico * 3 <= tamFisico) //Esperar a 1/3 del tamaño máximo para disminuir el tamaño fisico del vector
249 alojarMitad();
250
251 if(pos == UINT_MAX) //si pos == UINT_MAX entonces se borra por el final
252 pos = tamLogico;
253
254 T aux = vectorDin[pos -= 1]; //Tomo el valor indicado en la posición
255
256 for(unsigned int i = pos; i < tamLogico; i++){ //Itera hasta el último para ir trasladando elementos
257 vectorDin[i] = vectorDin[i+1];
258 }
259
260 tamLogico -= 1;
261 return aux;
262}
